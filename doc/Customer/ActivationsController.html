def show
  @user = User.find_by_perishable_token(params[:a])
  if @user && (@user.active? || @user.activate!)
    UserSession.create(@user, true)
    flash[:notice] = "Welcome back #{@user.name}"
  else
    flash[:notice] = "Invalid Activation URL!"
  end
  redirect_to root_url
end
