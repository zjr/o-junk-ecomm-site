def self._parse_dot(str, e) # :nodoc:
  if str.sub!(%r|('?-?\d+)\.\s*('?\d+)\.\s*('?-?\d+)|, ' ') # '
    case Format::STYLE[:dot]
    when :mdy
      s3e(e, $3, $1, $2)
    when :dmy
      s3e(e, $3, $2, $1)
    else
      s3e(e, $1, $2, $3)
    end
    true
  end
end
