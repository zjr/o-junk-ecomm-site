def create
  @credit_card = session_admin_cart[:user].credit_cards.new(params[:credit_card])

  respond_to do |format|
    if @credit_card.save
      format.html { redirect_to(@credit_card, :notice => 'Credit card was successfully created.') }
    else
      format.html { render :action => "new" }
    end
  end
end
