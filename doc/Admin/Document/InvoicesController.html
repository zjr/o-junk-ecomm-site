def index
  params[:page] ||= 1
  params[:rows] ||= 25

def show
  @invoice = Invoice.includes([:order => [
                                          :bill_address,
                                          :ship_address
                                          ]]).find(params[:id])

  respond_to do |format|
    format.html
    format.pdf do
      #prawnto :prawn=>{:skip_page_creation=>true}
      send_data print_form(@invoice).render, :filename => "invoice_#{@invoice.number}.pdf", :type => 'application/pdf'
    end
  end
end
